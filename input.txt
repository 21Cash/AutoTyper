#include<stdio.h>
void main()
{
	int m, n, p, q, i, j, k, sum = 0;
	printf("Enter the row & column sizes of matrix-1 : ");
	scanf("%d%d",&m, &n);
	printf("Enter the row & column sizes of matrix-2 : ");
	scanf("%d%d",&p, &q);
	if(n != p)
		printf("Multiplication is not possible\n");
	else
	{	
		int a[m][n], b[p][q], c[m][q];
		printf("Enter matrix-1 %d elements : ", m*n);
		for(i = 0; i < m; i++)
		{
			for(j = 0; j < n; j++)
				scanf("%d",&a[i][j]);
		}
		printf("Enter matrix-2 %d elements : ", p*q);
		for(i = 0; i < p; i++)
		{
			for(j = 0; j < q; j++)
				scanf("%d",&b[i][j]);
		}
		printf("The given matrix-1 is\n");
		for(i = 0; i < m; i++)
		{
			for(j = 0; j < n; j++)
				printf("%d ",a[i][j]);
			printf("\n");
		}
		printf("The given matrix-2 is\n");
		for(i = 0; i < p; i++)
		{
			for(j = 0; j < q; j++)
				printf("%d ",b[i][j]);
			printf("\n");	
		}
		printf("Multiplication of two matrices is\n");
		for(i = 0; i < m; i++)
		{
			for(j = 0; j < q; j++)
			{
				for(k = 0; k < n; k++)
					sum += a[i][k] * b[k][j];
				c[i][j] = sum;
				sum = 0;
			}
		}
		for(i = 0; i < m; i++)
		{
			for(j = 0; j < q; j++)
				printf("%d ", c[i][j]);
			printf("\n");
		}
	}
}
